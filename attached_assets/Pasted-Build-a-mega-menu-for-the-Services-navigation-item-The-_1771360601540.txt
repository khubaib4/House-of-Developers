interface MegaMenuProps {
  onClose?: () => void;
}

export function MegaMenu({ onClose }: MegaMenuProps) {
  const [isOpen, setIsOpen] = useState(false);
  const [isMobile, setIsMobile] = useState(false);
  const menuRef = useRef<HTMLDivElement>(null);
  const timeoutRef = useRef<ReturnType<typeof setTimeout>>();

  // Detect mobile
  useEffect(() => {
    const checkMobile = () => {
      setIsMobile(window.innerWidth < 768);
    };
    checkMobile();
    window.addEventListener("resize", checkMobile);
    return () => window.removeEventListener("resize", checkMobile);
  }, []);

  // Click outside to close (mobile)
  useEffect(() => {
    if (!isMobile) return;

    const handleClickOutside = (event: MouseEvent) => {
      if (
        menuRef.current &&
        !menuRef.current.contains(event.target as Node)
      ) {
        setIsOpen(false);
        onClose?.();
      }
    };

    if (isOpen) {
      document.addEventListener("mousedown", handleClickOutside);
    }

    return () => {
      document.removeEventListener("mousedown", handleClickOutside);
    };
  }, [isOpen, isMobile, onClose]);

  const handleMouseEnter = () => {
    if (isMobile) return; // Desktop only
    if (timeoutRef.current) {
      clearTimeout(timeoutRef.current);
    }
    setIsOpen(true);
  };

  const handleMouseLeave = () => {
    if (isMobile) return; // Desktop only
    timeoutRef.current = setTimeout(() => {
      setIsOpen(false);
    }, 150); // Small delay before closing
  };

  const handleClick = () => {
    if (!isMobile) return; // Mobile only
    setIsOpen(!isOpen);
  };

  const handleLinkClick = () => {
    setIsOpen(false);
    onClose?.();
  };

  return (
    <div
      ref={menuRef}
      className="relative"
      onMouseEnter={handleMouseEnter}
      onMouseLeave={handleMouseLeave}
    >
      {/* Trigger Button */}
      <button
        onClick={handleClick}
        className="
          flex items-center gap-1
          px-3 py-2
          text-sm font-medium
          text-foreground
          hover:text-primary
          transition-colors
          duration-200
        "
        aria-expanded={isOpen}
        aria-haspopup="true"
      >
        Services
        <ChevronRight
          size={14}
          className={`
            transition-transform duration-200
            ${isOpen ? "rotate-90" : ""}
          `}
        />
      </button>

      {/* Mega Menu Panel */}
      <AnimatePresence>
        {isOpen && (
          <motion.div
            initial={{ opacity: 0, y: -10 }}
            animate={{ opacity: 1, y: 0 }}
            exit={{ opacity: 0, y: -10 }}
            transition={{ duration: 0.2, ease: "easeOut" }}
            className="
              absolute
              left-1/2
              -translate-x-1/2
              top-full
              mt-2
              w-screen
              max-w-6xl
              z-50
              md:left-0
              md:translate-x-0
            "
          >
            <div
              className="
                bg-background
                border
                rounded-2xl
                shadow-2xl
                overflow-hidden
              "
              style={{
                backdropFilter: "blur(24px) saturate(180%)",
                WebkitBackdropFilter: "blur(24px) saturate(180%)",
                background: "rgba(var(--background), 0.95)",
              }}
            >
              {/* Content */}
              <div className="p-6 md:p-8">
                {/* 5 Column Grid */}
                <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-5 gap-8">
                  {serviceCategories.map((category) => (
                    <div key={category.name} className="space-y-4">
                      {/* Category Header */}
                      <Link href={category.href} onClick={handleLinkClick}>
                        <div className="flex items-center gap-3 group cursor-pointer">
                          {/* Icon Circle */}
                          <div
                            className={`
                              w-10 h-10
                              rounded-xl
                              flex items-center justify-center
                              bg-gradient-to-br ${category.color}
                              group-hover:scale-110
                              transition-transform duration-200
                            `}
                          >
                            <category.icon size={18} className="text-white" />
                          </div>

                          {/* Category Name */}
                          <div>
                            <h3
                              className="
                                font-semibold text-sm
                                text-foreground
                                group-hover:text-primary
                                transition-colors
                              "
                            >
                              {category.name}
                            </h3>
                          </div>
                        </div>
                      </Link>

                      {/* Sub-Services List */}
                      <ul className="space-y-2.5">
                        {category.subServices.map((service) => (
                          <li key={service.href}>
                            <Link href={service.href} onClick={handleLinkClick}>
                              <div
                                className="
                                  group
                                  flex items-center justify-between
                                  text-sm
                                  text-muted-foreground
                                  hover:text-primary
                                  transition-colors
                                  cursor-pointer
                                "

                                  {service.name}

                                    group-hover:opacity-100
                                    group-hover:translate-x-0.5
                                    transition-all
                                    duration-200
                                  "
                                />
                              </div>
                            </Link>
                          </li>
                        ))}
                      </ul>
                    </div>
                  ))}
                </div>

                {/* Bottom CTA */}
                <div
                  className="
                    mt-8
                    pt-8
                    border-t
                    flex flex-col sm:flex-row
                    items-center
                    justify-between
                    gap-4
                  "
                >
                  <div>
                    <p className="text-sm font-semibold text-foreground">
                      Not sure which service you need?
                    </p>
                    <p className="text-xs text-muted-foreground mt-1">
                      Book a free consultation and we'll recommend the right solution

                      className="
                        flex items-center gap-2
                        px-6 py-2.5
                        bg-primary
                        hover:bg-primary/90
                        text-primary-foreground
                        text-sm font-semibold
                        rounded-xl
                        transition-all
                        duration-200
                        active:scale-95
                        whitespace-nowrap
                      "
                    >
                      Get Free Consultation
                      <ArrowRight size={14} />
                    </button>
                  </Link>
                </div>
              </div>
            </div>
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
}

- Full-width up to max-w-6xl
- Positioned: absolute, centered under nav
- Background: frosted glass (backdrop-blur-24px)
- Border + shadow-2xl
- Rounded-2xl corners
- Padding: p-6 md:p-8

Category Header:
- Icon: 40px circle with gradient (each category has unique color)
- Name: font-semibold, text-sm
- Hover: icon scales to 110%, name turns primary color
- Clickable: links to category hub page

Sub-Service Links:
- text-sm, text-muted-foreground
- Hover: text-primary color
- ChevronRight icon appears on hover (slides in from right)
- Popular badge: "POPULAR" pill (primary/10 bg, primary text)

Bottom CTA:
- Border-top separator

- Responsive: stacks vertical on mobile

Behavior:
- Desktop: Hover to open, small delay (150ms) before closing on mouse leave

- AnimatePresence for clean mount/unmount

MOBILE BEHAVIOR:

Mobile (< 768px):
- Click "Services" button to toggle menu
- Menu appears below button (still full-width, centered)
- Click any link → menu closes
- Click outside menu → menu closes
- Grid collapses to 1-2 columns for easy scrolling

Desktop (≥ 768px):
- Hover over "Services" → menu opens
- Move mouse away → menu closes after 150ms
- Click any link → menu closes
- Hover interactions smooth and responsive
